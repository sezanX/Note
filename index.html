int stack[100], n=100, top=-1;
void push(int x)
{
    if(top>=n-1)
    {printf("stack overflow");}
        
    else
    {
        top=top+1;
        stack[top]=x;
    }
    
}
void pop()
{
    
    if(top==-1)
    {printf("stack underflow");}
        
    else
    {
        top=top-1;
    }
    
}

void IsEmpty()
{
    if(top==-1)
    {printf("Empty");}
    else
    {printf("Not Empty");}
    
}
